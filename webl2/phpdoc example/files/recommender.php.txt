/**
 * Get name recommendations.
 * @param string $gender The gender of the names
 * @param int $userId The id of the user for whom to provide recommendations
 * @param int $number The number of names to recommend
 * @return string[] The selected names
 */
function getRecommendations($gender, $userId, $number){
	// get nearest recommendations based on nearest neihgbors
	$recommendations = knn($gender, $userId, $number);

	// Get additional random names
	$missing = $number - sizeof($recommendations);
	$additionalRecommendations = getRandomNames($gender, $userId, $missing);

	// Merge and return
	return array_merge($recommendations, $additionalRecommendations);
}

/**
 * Get name recommendations based on the nearest neighbors.
 * @param string $gender the gender of the names
 * @param int $userId the id of the user for whom to provide recommendations
 * @param int $number the number of names to recommend
 * @return string[] the selected names
 */
function knn($gender, $userId, $number){
	// Maximum number of neighbors
	$k = 10;

	// Find similar users
	$neighbors = getNeighbors($userId, $k);

	// An array to store the scores of the candidate names
	$candidates = array();

	// Browse the similar users to add the names
	foreach ($neighbors as $neighbor => $similarity) {
	
		// Score of the names from this neighbor
		$neighborCandidates = computeNeighborCandidates($neighbor);

		// Incorporation into the final list
		foreach ($neighborCandidates as $name => $count){
			if (isset($candidates[$name])){
				$previousScore = $candidates[$name];
				$candidates[$name] = $previousScore + $similarity * $count;
			}
			else{
				$candidates[$name] = $similarity * $count;
			}
		}
	}

	// Pick the indicated number of names at random
	$recommendations = array();
	while(sizeof($recommendations) < $number && sizeof($candidates) > 0){

		// Pick a name
		do{
			$nameId = selectRandomName($candidates);
			$name = getName($nameId);
		} while (in_array($name, $recommendations));

		// Check the name has not been compared yet and has the proper gender
		if (!hasBeenCompared($userId, $nameId) && gender($nameId) == $gender){
			$recommendations[] = $name;
		}

		// Remove from the candidates
		unset($candidates[$nameId]);
	}
	return $recommendations;
}

/**
 * A method to compute candidate names from a given neighbor.
 * The set of names are first extracted (local candidates), and 
 * for each of these names the score is updated.
 * The score of a name is computed based on the number of times it was prefered.
 * @param int $neighbor the neighbor
 * @return array The local candidates. key => value pairs, where key is a name id and value is the score for the name.
 */
function computeNeighborCandidates($neighbor){
	global $connection;

	// The array of candidate names to return
	$candidates = array();

	// Preferences of the neighbor
	$query = "SELECT * FROM userspreferences WHERE user_id=$neighbor";
	$result = $connection->query($query);
	if ($result){
		$preferences = $result->fetchAll(PDO::FETCH_OBJ);

		// Count the number of times each name was preferred
		for ($i = 0; $i < sizeof($preferences); $i++){
			$name = $preferences[$i]->first_name;
			if (isset($candidates[$name])){
				$candidates[$name] = $candidates[$name] + 1;
			}
			else{
				$candidates[$name] = 1;
			}
		}
	}
	return $candidates;
}

/**
 * Get similar users (based on their preferences)
 * @param int $userId the id of the user for whom to find similar users
 * @param int $k the number of neighbors
 * @return array the neighbors. key => value pairs, where key is user_id and value is the similarity.
 */
function getNeighbors($userId, $k){
	global $connection;

	// The array of neighbors to return
	$neighbors = array();

	// Preferences of the user
	$queryPrefUser = "SELECT * FROM userspreferences WHERE user_id=:user_id";
	$statementPrefUser = $connection->prepare($queryPrefUser);
	$statementPrefUser->bindValue(":user_id", $userId, PDO::PARAM_INT);
	$statementPrefUser->execute();
	$prefUser = $statementPrefUser->fetchAll(PDO::FETCH_ASSOC);

	// Get other users
	$otherUsers = array();
	$queryotherUsers = "SELECT id FROM users WHERE id <> :user_id";
	$statementOtherUsers = $connection->prepare($queryotherUsers);
	$statementOtherUsers->bindValue(":user_id", $userId, PDO::PARAM_INT);
	$statementOtherUsers->execute();
	$otherUsers = $statementOtherUsers->fetchAll(PDO::FETCH_ASSOC);
	
	// Preferences of the other users
	$queryPrefOtherUser = "SELECT * FROM userspreferences WHERE user_id=:user_id";
	$statementPrefOtherUser = $connection->prepare($queryPrefOtherUser);
	for($i = 0; $i < sizeof($otherUsers); $i++){
		$otherUserId = $otherUsers[$i]["id"];
		$statementPrefOtherUser->bindValue(":user_id", $otherUserId, PDO::PARAM_INT);
		$statementPrefOtherUser->execute();
		$prefOtherUser = $statementPrefOtherUser->fetchAll(PDO::FETCH_ASSOC);
		
		// Compute similarity and add the neighbor to the array
		$similarity = similarity($prefUser, $prefOtherUser);
		if ($similarity > 0){
			$neighbors[$otherUserId] = $similarity;
		}
	}

	// return the k top elements
	return topElements($neighbors, $k);
}

/**
 * Get the elements with the highest values
 * @param array $array An associative array of elements with numerical values.
 * @param int $k The number of elements to get.
 * @return array The resulting associative array.
 */
function topElements($array, $k){
	$result = array();
	do{
		$maxKey = NULL;
		$maxValue = -1;
		foreach ($array as $key => $value){
			if ($value > $maxValue){
				$maxKey = $key;
				$maxValue = $value;
			}
		}
		$result[$maxKey] = $maxValue;
		unset($array[$maxKey]);
		if (sizeof($array) == 0){
			return $result;
		}
	} while (sizeof($result < $k));
	return $result;
}

/**
 * Binary cosine between two user preference vectors (arrays)
 * @param array $prefUser the preferences of the first user
 * @param array $prefOtherUser the preferences of the other user
 * @return float the similarity
 */
function similarity($prefUser, $prefOtherUser){
	// Skip if one of the arrays is empty
	if (sizeof($prefUser) == 0 || sizeof($prefOtherUser) == 0){
		return 0;
	}

	// Count the number of common preferences
	$count = 0;
	for ($i = 0; $i < sizeof($prefUser); $i++){
		for ($j = 0; $j < sizeof($prefOtherUser); $j++){
			if ($prefUser[$i]["first_name"] == $prefOtherUser[$j]["first_name"]
				&& $prefUser[$i]["second_name"] == $prefOtherUser[$j]["second_name"]){
				$count++;
			}
		}
	}

	// Compute de binary cosine similarity from that
	$denominator = sqrt(sizeof($prefUser)) * sqrt(sizeof($prefOtherUser));
	return $count / $denominator;
}

/**
 * Pick names at random with a probability that is dependent on the popularity of the name. The names must have been not yet compared by the user.
 * @param string $gender the gender of the names
 * @param int $userId the id of the user
 * @param int $number the number of names to pick
 * @return string[] the selected names
 */
function getRandomNames($gender, $userId, $number){
	// Fetch monstruous global variables at the end of the code
	global $topFemaleNames;
	global $femaleSum;
	global $topMaleNames;
	global $maleSum;

	// Process gender
	$names = null;
	$sum = null;
	if ($gender == 'm'){
		$names = $topMaleNames;
		$sum = $maleSum;
	}
	else{
		$names = $topFemaleNames;
		$sum = $femaleSum;
	}

	// The array of recommendations to return
	$recommendations = array();

	// Pick the names
	while(sizeof($recommendations) < $number){
		do{
			$name = selectRandomName($names);
		} while (hasBeenCompared($name, $userId) || in_array($name, $recommendations));
		$recommendations[] = $name;
	}
	return $recommendations;
}

/**
 * Check if a name has already been compared to some other name by a user
 * @param int|string $name the name
 * @param int $userId the user id
 * @return int the id of the name 
 */
function hasBeenCompared($name, $userId){
	global $connection;

	// $name may be a string, in which case the corresponding id is retrieved
	if (is_string($name)){
		$name = getNameId($name);

		// If not in database, then return true (skip it)
		if ($name == NULL){
			return true;
		}
	}

	// Make the query
	$query = "SELECT * FROM userspreferences WHERE user_id = $userId AND (first_name = $name OR second_name = $name)";
	$result = $connection->query($query)->fetch(PDO::FETCH_OBJ);
	return $result;
}

/**
 * Get the gender of a name
 * @param int $nameId The name id
 * @return string The gender ('m' or 'f')
 */
function gender($nameId){
	global $connection;
	$query = "SELECT * FROM names WHERE id = $nameId";
	$result = $connection->query($query)->fetch(PDO::FETCH_OBJ);
	return $result->gender;
}

/**
 * Get the id of a name
 * @param string $name The name
 * @return int The id of the name 
 */
function getNameId($name){
	global $connection;
	$query = "SELECT id FROM names WHERE name = '$name'";
	$result = $connection->query($query)->fetch(PDO::FETCH_OBJ);
	if ($result){
		return $result->id;
	}
	else return NULL;
}

/**
 * Get a name from its id
 * @param int $id The name id
 * @return string The name 
 */
function getName($id){
	global $connection;
	$query = "SELECT name FROM names WHERE id = $id";
	$result = $connection->query($query)->fetch(PDO::FETCH_OBJ);
	if ($result){
		return $result->name;
	}
	else return NULL;
}

/**
 * Pick a name at random with a probability that is dependent on its popularity.
 * @param array $names An array containing the names and their popularities
 * @return string The selected name
 */
function selectRandomName($names){
	// Sum of values
	$sum = 0;
	foreach ($names as $key => $value) {
		$sum += $value;
	}

	// Pick an element
	$r = rand(0, $sum);
	$otherSum = 0;
	foreach ($names as $key => $value){
		$otherSum += $value;
		if ($otherSum > $r){
			return $key;
		}
	}
}
